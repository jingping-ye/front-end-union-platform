# 组件类型

该章节假设有以下两个子组件:

//	子组件 FirstComp.vue
<template>
  <div>
    <input v-model="str" />
  </div>
</template>

<script>
export default {
  data() {
    return {
      str: "",
    };
  },
};
</script>

<style></style>


//	子组件 SecondComp.vue
<template>
  <div>
    <div>
      {{ isShow ? "显示" : "隐藏" }}
    </div>
    <button @click="clickMe">clickMe</button>
  </div>
</template>

<script>
export default {
  data() {
    return {
      isShow: false,
    };
  },
  methods: {
    clickMe() {
      this.isShow = !this.isShow;
    },
  },
};
</script>

# 一、按状态划分

按状态可分为有状态组件和无状态组件

# 二、按是否动态划分

按是否动态分为动态组件和非动态组件；

# 1. 动态组件

  • 通过 :is控制，采用的v-if的机制控制
  • 因为使用v-if控制，所以组件内部的状态不能保存。要保留组件的状态，请使用keep-alive。详情参考keep-alive
// 父组件 index.vue
<template>
  <div>
    <button @click="componentName = 'FirstComp'">组件一</button>
    <button @click="componentName = 'SecondComp'">组件二</button>
    <component :is="componentName"></component>
  </div>
</template>

<script>
import FirstComp from "../components/FirstComp";
import SecondComp from "../components/SecondComp";
export default {
  components: {
    FirstComp,
    SecondComp,
  },
  data() {
    return {
      componentName: "FirstComp",
    };
  },
};
</script>

# 三、按是否同异步划分

按异步和同步可分为异步组件和同步组件；我们大多数时候使用的都是同步组件；如果加载组件的时候要控制组件的加载状态，可以使用异步组件

  • 加载组件的参数是reject，则拒绝加载组件。

异步组件演示如下：

<template>
  <div>
    <button @click="componentName = 'FirstComp'">组件一</button>
    <button @click="componentName = 'SecondComp'">组件二</button>
    <keep-alive include="FirstComp,SecondComp">
      <component :is="componentName"></component>
    </keep-alive>
  </div>
</template>

<script>
export default {
  components: {
    FirstComp: function(reslove) {
      require(["../components/FirstComp"], reslove);
    },
    SecondComp: () => import("../components/SecondComp"),
    ThirdComp: function(reslove) {
      setTimeout(() => {
        require(["../components/SecondComp"], reslove);
      }, 1000);
    },
    ForthComp: function(reslove) {
      setTimeout(() => {
        require(["../components/SecondComp"], reject);
      }, 1000);
    },
    FifthComp: () => ({
      // 需要加载的组件 (应该是一个 `Promise` 对象)
      component: import("../components/SecondComp"),
      // 异步组件加载时使用的组件
      loading: LoadingComponent,
      // 加载失败时使用的组件
      error: ErrorComponent,
      // 展示加载时组件的延时时间。默认值是 200 (毫秒)
      delay: 1000,
      // 如果提供了超时时间且组件加载也超时了，
      // 则使用加载失败时使用的组件。默认值是：`Infinity`
      timeout: 3000,
    }),
  },
  data() {
    return {
      componentName: "FirstComp",
    };
  },
};
</script>
