# 2. 文件处理

# 2.1 请求图片并下载(二进制流)

注意：请求图片时不要启用Mock.js，Mock.js中设置了responseType为空，将导致返回的文件数据为一堆乱码，而非blob对象。

/***
 * 通过axios下载图片
 */
downloadImgByAxios() {
  axios
    .get("http://localhost:8890/file/img", {
      responseType: "blob",
      headers: {
        "Content-Type": "image/png",
      },
    })
    .then((res) => {
      this.downloadByBlob(res.data, "cicd_flow.png", { type: "image/png" });
    })
    .catch((err) => {
      console.log("err", err);
    });
},
downloadByBlob(file, fileName = "", args = { type: "application/json;charset=utf-8" }) {
  let blob = new Blob([file], args);
  if (window.navigator && window.navigator.msSaveOrOpenBlob) {
    window.navigator.msSaveOrOpenBlob(blob, fileName);
  } else {
    // for Non-IE
    let objectUrl = URL.createObjectURL(blob);
    let link = document.createElement("a");
    link.href = objectUrl;
    link.setAttribute("download", fileName);
    document.body.appendChild(link);
    link.click();
    window.URL.revokeObjectURL(link.href);
  }
},

# 3. 特殊请求

# 3.1 同时发起多个请求

考虑如下场景，你需要在发起N个请求后，在全部请求完毕后处理信息。其中，N个请求均是独立的，没有上下关联。一种方法是进行串行，在.then中连续调用请求，一直到最后一个请求。这种方法可以实现最终目的，但是也存在缺点。一是串行链过长，代码变得复杂，很难看懂。二是语义上不明确，在.then中连续调用请求，这实际上表明了上下请求有链接关系，然而这里我们的所有请求都是明确的。下面我们介绍针对这种情况的处理方法。

假设我们现在有两个接口，在页面初始化时调用，调用过程中页面显示Loading，一直到两个接口全部查询完毕后，才显示页面信息。

# 3.1.1 Promise.all--promise.chain

下面这种方面采用的是promise链的方式

/**
 * HTTP REQUEST
 * 获取用户列表
 */
getUserListRequest() {
  return axios.get("https://www.fastmock.site/mock/cb28abad7a23ac4309b07f4072fcafdd/axios/user/list");
},
/***
 * HTTP REQUEST
 * 获取项目列表
 */
getProjectListRequest() {
  return axios.get("https://www.fastmock.site/mock/cb28abad7a23ac4309b07f4072fcafdd/axios/project/list");
},
getPromiseAll() {
  Promise.all([this.getUserListRequest(), this.getProjectListRequest()])
    .then(([{ data: userListRes }, { data: projectListRes }]) => {
      console.log("userListRes", userListRes);
      console.log("projectListRes", projectListRes);
    })
    .catch((err) => {
      console.log("err", err);
    });
},

# 3.2.2 无截止

无截止场景：每隔n秒刷新一个API，无间断。

这里说明每隔10秒刷新一次的场景。

data(){
  this.taskTimer = null;
  this.countdownTimer = null;
  return {
      countdown:10,
  }  
},
methods:{
 /*
  * 停止并回收定时器
  */
  clearAndCollectTimer(){
      clearInterval(this.countdownTimer);
      clearTimeout(this.taskTimer);
      this.countdownTimer=null;
      this.taskTimer=null;
  },
  /*
  * 设置倒计时
  */
  resetCountDownInterval(){
      this.countdown = 10;
      this.countdownTimer = setInterval(() => {
        if (this.countdown <= 1) {
          clearInterval(this.countdownTimer);
        } else {
          this.countdown = --this.countdown;
        }
      }, 1000);
  }
},
/*
* 10秒后调用
*/
refreshNotifyInterval(){
  this.taskTimer = setTimeout(()=>{
      this.getNotifyRequest();
  },10000);  
},
/*
* HTTP REQUEST
* @note: 10秒刷新一次提示信息
*/
getNotifyRequest(){
    this.clearAndCollectTimer();
    axios.get("/notify")
    .then(res=>{
        console.log("res",res);
    }).catch(err=>{
        console.log("err",err);
    }).finally(()=>{
        this.refreshNotifyInterval();
        this.resetCountDownInterval();
    })
},
mounted(){
   this.getNotifyRequest();
},
destroyed(){
   this.clearAndCollectTimer();   
}

# 4. 进度条

# 4.1 下载进度条

<template>
    <div>
      <a-space>
        <span>进度条：</span>
        <a-button type="primary" @click="downloadZipFile">下载629MB压缩包文件并显示进度条</a-button>
      </a-space>
      <div style="padding:10px;">
        下载进度：<a-progress
          :stroke-color="{
            from: '#108ee9',
            to: '#87d068',
          }"
          :percent="downLoadPercent"
          :status="getDownloadStatus"
        />
      </div>
    </div>
</template>
<script>
import "axios" from "axios";
export default {
    data(){
        downLoadPercent:0, // 下载进度
    },
    computed: {
        getDownloadStatus() {
          if (this.downLoadPercent == 0) {
            return "normal";
          } else if (this.downLoadPercent < 100) {
            return "active";
          } else if (this.downLoadPercent == 100) {
            return "success";
          } else {
            return "exception";
          }
        },
  	},
   methods:{
     downloadZipFile() {
          const _this = this;
          axios
            .get("http://localhost:8890/file/zip", {
              responseType: "blob",
              headers: {
                "Content-Type": "application/x-7z-compressed",
              },
              // 下载进度事件
              onDownloadProgress(file) {
                let percent = parseInt(100 * (file.loaded / file.total));
                _this.$nextTick(() => {
                  _this.downLoadPercent = percent;
                });
              },
            })
            .then((res) => {
              console.log("===res===", res);
              this.downloadByBlob(res.data, "书籍.zip", {
                type: "application/x-7z-compressed",
              });
            })
            .catch((err) => {
              console.log("err", err);
            });
        },
   }
}
</script>

# 5. 中大型项目的封装

封装考虑的问题：

  • 请求头部
    • 认证
    • cookie
    • 请求体类型 content-type
  • 请求体
    • 格式转换
  • 响应体
    • 状态码处理
      • 正确状态码
      • 错误状态码
    • 返回的业务定义格式 --> 业务逻辑判断
    • 格式转换
  • 其他配置
    • 代理
    • 超时时间
    • 多环境切换
    • reponseType
    • 打印请求和响应信息

这里我们假定项目的业务数据返回形式为：

{
    code: 2000,
    res:{
        info:[], 
        msg:""
    }
}

封装如下：

/**
 * axios请求封装
 */
import axios from "axios";
import { message } from "ant-design-vue";
import { jsonToHump, jsonToUnderline } from "./httpFormTran"; // 用来从驼峰转为下划线；下划线转驼峰
import { getToken, removeToken } from "./auth";

const isProxy = process.env.VUE_APP_API_PROXY === "1";
console.log("当前是否处于代理服务器下：", isProxy);

// 基础路径
let BaseURL = process.env.VUE_APP_API_BASE_URL;
if (isProxy) {
  BaseURL = "/proxy-service";
} else {
  BaseURL = "";
}

const Timeout = 30000; // 超时时间
const NormalCode = 4000; // 从后台正常返回的代码

const ErrorMessage = (msg) => {
  message.error(msg, 5);
}; // 错误信息提示

let httpInstance = axios.create({
  baseURL: BaseURL,
  timeout: Timeout,
});

//  请求表单
httpInstance.postForm = (url, data, config) => {
  return httpInstance.post(url, data, {
    headers: {
      "Content-Type": "multipart/form-data",
      Authorization: `Bearer ${getToken()}`,
    },
    ...config,
  });
};

//  请求拦截器
httpInstance.interceptors.request.use(
  (config) => {
    //  转换post请求数据
    if (config.data && Object.keys(config.data).length > 0) {
      config.data = jsonToUnderline(config.data);
    }

    //  转换query请求参数
    if (config.params && Object.keys(config.params).length > 0) {
      config.params = jsonToUnderline(config.params);
    }

    //  添加头部信息
    config.headers["Content-Type"] = "application/json;charset=UTF-8";

    //  添加token
    if (getToken()) {
      config.headers.Authorization = `Bearer ${getToken()}`;
    }

    console.log("request", config);
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

//  响应拦截器
httpInstance.interceptors.response.use(
  (response) => {
    console.log("response", response);
    if (response.status === 200) {
      if (response.data.res) {
        //  处理正常json格式文件
        response.data = jsonToHump(response.data); // 转为小驼峰
        if (response.data.code === NormalCode) {
          return Promise.resolve(response.data.info);
        } else {
          ErrorMessage(`错误(错误代码${response.data.code}):${response.data.msg}`);
          return Promise.reject(new Error("请求失败"));
        }
      } else {
        //  处理文件流
        return Promise.resolve(response.data);
      }
    } else {
      console.log(response);
    }
  },
  (error) => {
    if (error.response) {
      const resStatus = error.response.status.toString();
      if (resStatus === "401") {
        ErrorMessage(`token验证失败，请重新登录!`);
        removeToken();
        location.reload();
      } else if (resStatus.startsWith("5")) {
        ErrorMessage(`服务器繁忙中，请稍后再试！`);
        return Promise.reject(error);
      } else {
        return Promise.reject(error);
      }
    } else {
      ErrorMessage(error);
      return Promise.reject(error);
    }
  }
);
export default httpInstance;
